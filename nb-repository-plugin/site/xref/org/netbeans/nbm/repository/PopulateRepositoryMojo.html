89  /**
90   * A goal for identifying NetBeans modules from the installation and populating the local
91   * repository with them. Optionally you can also deploy to a remote repository.
92   * <p>
93   * If you are looking for an existing remote repository for NetBeans artifacts, check out
94   * <a href="http://bits.netbeans.org/nexus/content/groups/netbeans/">http://bits.netbeans.org/nexus/content/groups/netbeans/</a>,
95   * it contains API artifacts for multiple releases.
96   * <a href="http://bits.netbeans.org/netbeans/trunk/maven-snapshot/">http://bits.netbeans.org/netbeans/trunk/maven-snapshot/</a>
97   * may also be used for <code>SNAPSHOT</code> artifacts if you wish to test development builds.
98   * </p><p>
99   * See this <a href="http://mojo.codehaus.org/nbm-maven/nbm-maven-plugin/repository.html">HOWTO</a> on how to generate the NetBeans binaries required
100  * by this goal.
101  * </p>
102  *
103  * @author <a href="mailto:mkleint@codehaus.org">Milos Kleint</a>
104  */
105 @Mojo(name="populate", aggregator=true, requiresProject=false)
106 public class PopulateRepositoryMojo
107     extends AbstractNetbeansMojo

109     private static final String GROUP_API = ".api";
110     private static final String GROUP_IMPL = ".modules";
111     private static final String GROUP_EXTERNAL = ".external";
112     private static final String GROUP_CLUSTER = ".cluster";

268     @Override
269     public void execute()
270         throws MojoExecutionException
271     {
272         getLog().info( "Populate repository with NetBeans modules" );
273         Project antProject = antProject();
274         ArtifactRepository deploymentRepository = null;
275         if ( deployUrl != null )
276         {
277             ArtifactRepositoryLayout layout = new DefaultRepositoryLayout();
278             deploymentRepository = repositoryFactory.createDeploymentArtifactRepository(
279                 "netbeans", deployUrl, layout, true );
280         }
281         else if ( skipLocalInstall )
282         {
283             throw new MojoExecutionException(
284                     "When skipping install to local repository, one shall define the deployUrl parameter" );
285         }
286 
287         IndexSearcher searcher = null;
288         if ( nexusIndexDirectory != null && nexusIndexDirectory.exists() )
289         {
290             try
291             {
292                 Directory nexusDir = FSDirectory.open( nexusIndexDirectory.toPath() );
293                 IndexReader nexusReader = DirectoryReader.open( nexusDir );
294                 searcher = new IndexSearcher( nexusReader );
295                 getLog().info( "Opened index with " + nexusReader.numDocs() + " documents" );
296             }
297             catch ( IOException ex )
298             {
299                 getLog().error( "Could not open " + nexusIndexDirectory, ex );
300             }
301         }
302 
303         if ( netbeansInstallDirectory == null )
304         {
305             Input input = (Input) antProject.createTask( "input" );
306             input.setMessage( "Please enter NetBeans installation directory:" );
307             input.setAddproperty( "installDir" );
308             try
309             {
310                 input.execute();
311             }
312             catch ( BuildException e )
313             {
314                 getLog().error( "Cannot run ant:input" );
315                 throw new MojoExecutionException( e.getMessage(), e );
316             }
317             String prop = antProject.getProperty( "installDir" );
318             netbeansInstallDirectory = new File( prop );
319         }
320 
321         File rootDir = netbeansInstallDirectory;
322         if ( !rootDir.exists() )
323         {
324             getLog().error( "NetBeans installation doesn't exist." );
325             throw new MojoExecutionException( "NetBeans installation doesn't exist." );
326         }
327         getLog().info( "Copying NetBeans artifacts from " + netbeansInstallDirectory );
328 
329         PathConvert convert = (PathConvert) antProject.createTask( "pathconvert" );
330         convert.setPathSep( "," );
331         convert.setProperty( "netbeansincludes" );
332         FileSet set = new FileSet();
333         set.setDir( rootDir );
334         set.createInclude().setName( "**/modules/*.jar" );
335         set.createInclude().setName( "*/core/*.jar" );
336         set.createInclude().setName( "platform*/lib/*.jar" );
337 
338         convert.createPath().addFileset( set );
339         try
340         {
341             convert.execute();
342         }
343         catch ( BuildException e )
344         {
345             getLog().error( "Cannot run ant:pathconvert" );
346             throw new MojoExecutionException( e.getMessage(), e );
347         }
348 
349         String prop = antProject.getProperty( "netbeansincludes" );
350         StringTokenizer tok = new StringTokenizer( prop, "," );
351         HashMap<ModuleWrapper, Artifact> moduleDefinitions = new HashMap<ModuleWrapper, Artifact>();
352         HashMap<String, Collection<ModuleWrapper>> clusters = new HashMap<String, Collection<ModuleWrapper>>();
353         while ( tok.hasMoreTokens() )
354         {
355             String token = tok.nextToken();
356             File module = new File( token );
357             String clust = module.getAbsolutePath().substring( rootDir.getAbsolutePath().length() + 1 );
358             clust = clust.substring( 0, clust.indexOf( File.separator ) );
359             ExamineManifest examinator = new ExamineManifest( getLog() );
360             examinator.setPopulateDependencies( true );
361             examinator.setJarFile( module );
362             examinator.checkFile();
363             if ( examinator.isNetBeansModule() || examinator.isOsgiBundle() )
364             {
365                 //TODO get artifact id from the module's manifest?
366                 String artifact = module.getName().substring( 0, module.getName().indexOf( ".jar" ) );
367                 if ( "boot".equals( artifact ) )
368                 {
369                     artifact = "org-netbeans-bootstrap";
370                 }
371                 if ( "core".equals( artifact ) )
372                 {
373                     artifact = "org-netbeans-core-startup";
374                 }
375                 if ( "core-base".equals( artifact ) )
376                 {
377                     artifact = "org-netbeans-core-startup-base";
378                 }                
379                 String version = forcedVersion == null ? examinator.getSpecVersion() : forcedVersion;
380                 String group = groupIdPrefix + (examinator.isOsgiBundle() ? GROUP_EXTERNAL : examinator.hasPublicPackages() ? GROUP_API : GROUP_IMPL);
381                 Artifact art = createArtifact( artifact, version, group );
382                 if ( examinator.isOsgiBundle() )
383                 {
384                     Dependency dep = findExternal( searcher, module );
385                     if ( dep != null )
386                     {
387                         // XXX use those coords instead of publishing this
388                         // (for now all bundles are from Orbit, which does not publish to Central, or specially built)
389                     }
390                 }
391                 ModuleWrapper wr = new ModuleWrapper( artifact, version, group, examinator, module );
392                 wr.setCluster( clust );
393                 moduleDefinitions.put( wr, art );
394                 Collection<ModuleWrapper> col = clusters.get( clust );
395                 if ( col == null )
396                 {
397                     col = new ArrayList<ModuleWrapper>();
398                     clusters.put( clust, col );
399                 }
400                 col.add( wr );
401             }
402         }
403         List<ModuleWrapper> wrapperList = new ArrayList<ModuleWrapper>( moduleDefinitions.keySet() );
404         int count = wrapperList.size() + 1;
405         int index = 0;
406         File javadocRoot = null;
407         if ( netbeansJavadocDirectory != null )
408         {
409             javadocRoot = netbeansJavadocDirectory ;
410             if ( !javadocRoot.exists() )
411             {
412                 javadocRoot = null;
413                 throw new MojoExecutionException(
414                     "The netbeansJavadocDirectory parameter doesn't point to an existing folder" );
415             }
416         }
417         File sourceRoot = null;
418         if ( netbeansSourcesDirectory != null )
419         {
420             sourceRoot = netbeansSourcesDirectory;
421             if ( !sourceRoot.exists() )
422             {
423                 sourceRoot = null;
424                 throw new MojoExecutionException(
425                     "The netbeansSourceDirectory parameter doesn't point to an existing folder" );
426             }
427         }
428 
429         File nbmRoot = null;
430         if ( netbeansNbmDirectory != null )
431         {
432             nbmRoot = netbeansNbmDirectory;
433             if ( !nbmRoot.exists() )
434             {
435                 nbmRoot = null;
436                 throw new MojoExecutionException(
437                     "The nbmDirectory parameter doesn't point to an existing folder" );
438             }
439         }
440 
441         List<ExternalsWrapper> externals = new ArrayList<ExternalsWrapper>();
442         try
443         {
444             for ( Map.Entry<ModuleWrapper, Artifact> elem : moduleDefinitions.entrySet() )
445             {
446                 ModuleWrapper man = elem.getKey();
447                 Artifact art = elem.getValue();
448                 index = index + 1;
449                 getLog().info( "Processing " + index + "/" + count );
450                 File pom = createMavenProject( man, wrapperList, externals, searcher );
451                 ArtifactMetadata metadata = new ProjectArtifactMetadata( art, pom );
452                 art.addMetadata( metadata );
453                 File javadoc = null;
454                 Artifact javadocArt = null;
455                 if ( javadocRoot != null )
456                 {
457                     File zip = new File( javadocRoot, art.getArtifactId() + ".zip" );
458                     if ( zip.exists() )
459                     {
460                         javadoc = zip;
461                         javadocArt = createAttachedArtifact( art, javadoc, "jar", "javadoc" );
462                     }
463                 }
464                 File source = null;
465                 Artifact sourceArt = null;
466                 if ( sourceRoot != null )
467                 {
468                     File zip = new File( sourceRoot, art.getArtifactId() + ".zip" );
469                     if ( zip.exists() )
470                     {
471                         source = zip;
472                         sourceArt = createAttachedArtifact( art, source, "jar", "sources" );
473                     }
474                 }
475                 File nbm = null;
476                 Artifact nbmArt = null;
477                 if ( nbmRoot != null )
478                 {
479                     File zip = new File( nbmRoot, art.getArtifactId() + ".nbm" );
480 
481                     if ( !zip.exists() )
482                     {
483                         zip = new File( nbmRoot,
484                             man.getCluster() + File.separator + art.getArtifactId() + ".nbm" );
485                     }
486                     if ( zip.exists() )
487                     {
488                         nbm = zip;
489                         nbmArt = createAttachedArtifact( art, nbm, "nbm-file", null );
490                         if ( nbmArt.getArtifactHandler().getExtension().equals( "nbm-file" ) )
491                         {
492                             // Maven 2.x compatibility.
493                             nbmArt = createAttachedArtifact( art, nbm, "nbm", null );
494                         }
495                         assert nbmArt.getArtifactHandler().getExtension().equals( "nbm" );
496                     }
497                 }
498                 File moduleJar = man.getFile();
499                 File moduleJarMinusCP = null;
500                 if ( ! man.getModuleManifest().getClasspath().isEmpty() )
501                 {
502                     try
503                     {
504                         moduleJarMinusCP = File.createTempFile( man.getArtifact(), ".jar" );
505                         moduleJarMinusCP.deleteOnExit();
506                         InputStream is = new FileInputStream( moduleJar );
507                         try
508                         {
509                             OutputStream os = new FileOutputStream( moduleJarMinusCP );
510                             try
511                             {
512                                 JarInputStream jis = new JarInputStream( is );
513                                 Manifest mani = new Manifest( jis.getManifest() );
514                                 mani.getMainAttributes().remove( Attributes.Name.CLASS_PATH );
515                                 if ( !man.deps.isEmpty() )
516                                 { // MNBMODULE-132
517                                     StringBuilder b = new StringBuilder();
518                                     for ( Dependency dep : man.deps )
519                                     {
520                                         if ( b.length() > 0 )
521                                         {
522                                             b.append( ' ' );
523                                         }
524                                         b.append( dep.getGroupId() ).append( ':' ).append( dep.getArtifactId() ).append( ':' ).append( dep.getVersion() );
525                                         if (dep.getClassifier() != null) {
526                                             b.append(":").append(dep.getClassifier());
527                                         }
528                                     }
529                                     mani.getMainAttributes().putValue( "Maven-Class-Path", b.toString() );
530                                 }
531                                 else
532                                 {
533                                     getLog().warn( "did not find any external artifacts for " + man.getModule() );
534                                 }
535                                 JarOutputStream jos = new JarOutputStream( os, mani );
536                                 JarEntry entry;
537                                 while ( ( entry = jis.getNextJarEntry() ) != null )
538                                 {
539                                     if ( entry.getName().matches( "META-INF/.+[.]SF" ) )
540                                     {
541                                         throw new IOException( "cannot handle signed JARs" );
542                                     }
543                                     jos.putNextEntry( entry );
544                                     byte[] buf = new byte[(int) entry.getSize()];
545                                     int read = jis.read( buf, 0, buf.length );
546                                     if ( read != buf.length )
547                                     {
548                                         throw new IOException( "read wrong amount" );
549                                     }
550                                     jos.write( buf );
551                                 }
552                                 jos.close();
553                             }
554                             finally
555                             {
556                                 os.close();
557                             }
558                         }
559                         finally
560                         {
561                             is.close();
562                         }
563                     }
564                     catch ( IOException x )
565                     {
566                         getLog().warn( "Could not process " + moduleJar + ": " + x, x );
567                         moduleJarMinusCP.delete();
568                         moduleJarMinusCP = null;
569                     }
570                 }
571                 try
572                 {
573                     if ( !skipLocalInstall )
574                     {
575                         install( moduleJarMinusCP != null ? moduleJarMinusCP : moduleJar, art );
576                         if ( javadoc != null )
577                         {
578                             install( javadoc, javadocArt );
579                         }
580                         if ( source != null )
581                         {
582                             install( source, sourceArt );
583                         }
584                         if ( nbm != null )
585                         {
586                             install( nbm, nbmArt );
587                         }
588                     }
589                     try
590                     {
591                         if ( deploymentRepository != null )
592                         {
593                             artifactDeployer.deploy( moduleJarMinusCP != null ? moduleJarMinusCP : moduleJar, art,
594                                                      deploymentRepository, localRepository );
595                             if ( javadoc != null )
596                             {
597                                 artifactDeployer.deploy( javadoc, javadocArt, deploymentRepository, localRepository );
598                             }
599                             if ( source != null )
600                             {
601                                 artifactDeployer.deploy( source, sourceArt, deploymentRepository, localRepository );
602                             }
603                             if ( nbm != null )
604                             {
605                                 artifactDeployer.deploy( nbm, nbmArt, deploymentRepository, localRepository );
606                             }
607                         }
608                     }
609                     catch ( ArtifactDeploymentException ex )
610                     {
611                         throw new MojoExecutionException( "Error Deploying artifact", ex );
612                     }
613                 }
614                 finally
615                 {
616                     if ( moduleJarMinusCP != null )
617                     {
618                         moduleJarMinusCP.delete();
619                     }
620                 }
621             }
622         }
623         finally
624         {
625             /*if ( searcher != null )
626             {
627                 try
628                 {
629                     searcher.close();
630                 }
631                 catch ( IOException ex )
632                 {
633                     getLog().error( ex );
634                 }
635             }*/
636         }
637 
638         //process collected non-recognized external jars..
639         if ( externals.size() > 0 )
640         {
641             index = 0;
642             count = externals.size();
643             for ( ExternalsWrapper ex : externals )
644             {
645                 Artifact art = createArtifact( ex.getArtifact(), ex.getVersion(), ex.getGroupid() );
646                 index = index + 1;
647                 getLog().info( "Processing external " + index + "/" + count );
648                 File pom = createExternalProject( ex );
649                 ArtifactMetadata metadata = new ProjectArtifactMetadata( art, pom );
650                 art.addMetadata( metadata );
651                 if ( !skipLocalInstall )
652                 {
653                     install( ex.getFile(), art );
654                 }
655                 try
656                 {
657                     if ( deploymentRepository != null )
658                     {
659                         artifactDeployer.deploy( ex.getFile(), art,
660                             deploymentRepository, localRepository );
661                     }
662                 }
663                 catch ( ArtifactDeploymentException exc )
664                 {
665                     throw new MojoExecutionException( "Error Deploying artifact", exc );
666                 }
667             }
668         }
669 
670         if ( ! defineCluster )
671         {
672             getLog().info( "Not creating cluster POMs." );
673         }
674         else if ( forcedVersion == null )
675         {
676             getLog().warn( "Version not specified, cannot create cluster POMs." );
677         }
678         else
679         {
680             for ( Map.Entry<String, Collection<ModuleWrapper>> elem : clusters.entrySet() )
681             {
682                 String cluster = stripClusterName( elem.getKey() );
683                 Collection<ModuleWrapper> modules = elem.getValue();
684                 getLog().info( "Processing cluster " + cluster );
685                 Artifact art = createClusterArtifact( cluster, forcedVersion );
686                 File pom = createClusterProject( art, modules );
687                 ProjectArtifactMetadata metadata = new ProjectArtifactMetadata( art, pom );
688                 art.addMetadata( metadata );
689                 if ( !skipLocalInstall )
690                 {
691                     install( pom, art );
692                 }
693                 try
694                 {
695                     if ( deploymentRepository != null )
696                     {
697                         artifactDeployer.deploy( pom, art, deploymentRepository, localRepository );
698                     }
699                 }
700                 catch ( ArtifactDeploymentException ex )
701                 {
702                     throw new MojoExecutionException( "Error Deploying artifact", ex );
703                 }
704             }
705 
706         }

746     private File createMavenProject( ModuleWrapper wrapper, List<ModuleWrapper> wrapperList,
747                                      List<ExternalsWrapper> externalsList, IndexSearcher searcher )
748             throws MojoExecutionException
749     {
750         Model mavenModel = new Model();
751 
752         mavenModel.setGroupId( wrapper.getGroup() );
753         mavenModel.setArtifactId( wrapper.getArtifact() );
754         mavenModel.setVersion( wrapper.getVersion() );
755         mavenModel.setPackaging( "jar" );
756         mavenModel.setModelVersion( "4.0.0" );
757         ExamineManifest man = wrapper.getModuleManifest();
758         List<Dependency> deps = new ArrayList<Dependency>();
759         if ( !man.getDependencyTokens().isEmpty() )
760         {
761             for ( String elem : man.getDependencyTokens() )
762             {
763                 // create pseudo wrapper
764                 ModuleWrapper wr = new ModuleWrapper( elem );
765                 int index = wrapperList.indexOf( wr );
766                 if ( index > -1 )
767                 {
768                     wr = wrapperList.get( index );
769                     Dependency dep = new Dependency();
770                     dep.setArtifactId( wr.getArtifact() );
771                     dep.setGroupId( wr.getGroup() );
772                     dep.setVersion( wr.getVersion() );
773                     dep.setType( "jar" );
774                     //we don't want the API modules to depend on non-api ones..
775                     // otherwise the transitive dependency mechanism pollutes your classpath..
776                     if ( wrapper.getModuleManifest().hasPublicPackages() && !wr.getModuleManifest().hasPublicPackages() )
777                     {
778                         dep.setScope( "runtime" );
779                     }
780                     deps.add( dep );
781                 }
782                 else if ( dependencyRepositoryUrl != null )
783                 {
784                     Dependency dep = new Dependency();
785                     dep.setType( "jar" );
786                     String artifactId = elem.replace( '.', '-' );
787                     dep.setArtifactId( artifactId );
788                     if ( forcedVersion == null )
789                     {
790                         throw new MojoExecutionException( "Cannot use dependencyRepositoryUrl without forcedVersion" );
791                     }
792                     dep.setVersion( forcedVersion );
793                     ArtifactRepositoryPolicy policy = new ArtifactRepositoryPolicy();
794                     List<ArtifactRepository> repos = Collections.singletonList(
795                             repositoryFactory.createArtifactRepository( dependencyRepositoryId, dependencyRepositoryUrl, artifactRepositoryLayout, policy, policy) );
796                     try
797                     {
798                         artifactResolver.resolve( artifactFactory.createBuildArtifact( groupIdPrefix + GROUP_API, artifactId, forcedVersion, "pom" ), repos, localRepository );
799                         dep.setGroupId( groupIdPrefix + GROUP_API );
800                     }
801                     catch ( AbstractArtifactResolutionException x )
802                     {
803                         try
804                         {
805                             artifactResolver.resolve( artifactFactory.createBuildArtifact( groupIdPrefix + GROUP_IMPL, artifactId, forcedVersion, "pom" ), repos, localRepository );
806                             dep.setGroupId( groupIdPrefix + GROUP_IMPL );
807                             if ( wrapper.getModuleManifest().hasPublicPackages() )
808                             {
809                                 dep.setScope( "runtime" );
810                             }
811                         }
812                         catch ( AbstractArtifactResolutionException x2 )
813                         {
814                             try
815                             {
816                                 artifactResolver.resolve( artifactFactory.createBuildArtifact( groupIdPrefix + GROUP_EXTERNAL, artifactId, forcedVersion, "pom" ), repos, localRepository );
817                                 dep.setGroupId( groupIdPrefix + GROUP_EXTERNAL );
818                                 if ( wrapper.getModuleManifest().hasPublicPackages() )
819                                 {
820                                     dep.setScope( "runtime" );
821                                 }
822                             }
823                             catch ( AbstractArtifactResolutionException x3 )
824                             {
825                                 getLog().warn( x3.getOriginalMessage() );
826                                 throw new MojoExecutionException( "No module found for dependency '" + elem + "'", x );
827                             }
828 
829                            
830                         }
831 
832                     }
833                     deps.add( dep );
834                 }
835                 else
836                 {
837                     getLog().warn( "No module found for dependency '" + elem + "'" );
838                 }
839             }
840         }
841         //need some generic way to handle Classpath: items.
842         //how to figure the right version?
843         String cp = wrapper.getModuleManifest().getClasspath();
844         if ( !cp.isEmpty() )
845         {
846             StringTokenizer tok = new StringTokenizer( cp );
847             while ( tok.hasMoreTokens() )
848             {
849                 String path = tok.nextToken();
850                 File f = new File( wrapper.getFile().getParentFile(), path );
851                 if ( f.exists() )
852                 {
853                     Dependency dep = findExternal( searcher, f );
854                     if ( dep != null )
855                     {
856                         deps.add( dep );
857                         // XXX MNBMODULE-170: repack NBM with *.external
858                     }
859                     else
860                     {
861                         ExternalsWrapper ex = new ExternalsWrapper();
862                         ex.setFile( f );
863                         String artId = f.getName();
864                         if ( artId.endsWith( ".jar" ) )
865                         {
866                             artId = artId.substring( 0, artId.length() - ".jar".length() );
867                         }
868                         ex.setVersion( wrapper.getVersion() );
869                         ex.setArtifact( artId );
870                         ex.setGroupid( groupIdPrefix + GROUP_EXTERNAL );
871                         externalsList.add( ex );
872                         dep = new Dependency();
873                         dep.setArtifactId( artId );
874                         dep.setGroupId( groupIdPrefix + GROUP_EXTERNAL );
875                         dep.setVersion( wrapper.getVersion() );
876                         dep.setType( "jar" );
877                         deps.add( dep );
878                     }
879                 }
880             }
881         }
882 
883         wrapper.deps = deps;
884         mavenModel.setDependencies( deps );
885         FileWriter writer = null;
886         File fil = null;
887         try
888         {
889             MavenXpp3Writer xpp = new MavenXpp3Writer();
890             fil = File.createTempFile( "maven", ".pom" );
891             fil.deleteOnExit();
892             writer = new FileWriter( fil );
893             xpp.write( writer, mavenModel );
894         }
895         catch ( IOException ex )
896         {
897             ex.printStackTrace();
898 
899         }
900         finally
901         {
902             if ( writer != null )
903             {
904                 try
905                 {
906                     writer.close();
907                 }
908                 catch ( IOException io )
909                 {
910                     io.printStackTrace();
911                 }
912             }
913         }
914         return fil;
915     }

917     private Dependency findExternal( IndexSearcher searcher, File f )
918     {
919         if ( searcher == null )
920         {
921             return null;
922         }
923         try
924         {
925             MessageDigest shaDig = MessageDigest.getInstance( "SHA1" );
926             InputStream is = new FileInputStream( f );
927             try {
928                 OutputStream os = new DigestOutputStream( new NullOutputStream(), shaDig );
929                 IOUtil.copy( is, os );
930                 os.close();
931             }
932             finally
933             {
934                 is.close();
935             }
936             String sha = encode( shaDig.digest() );
937             TermQuery q = new TermQuery( new Term( "1", sha ) );
938             TopScoreDocCollector collector = TopScoreDocCollector.create( 5 );
939             searcher.search(q, collector);
940             ScoreDoc[] hits = collector.topDocs().scoreDocs;
941             if ( hits.length >= 1 )
942             {
943                 int docId = hits[0].doc;    
944                 Document doc = searcher.doc(docId);                
945                 IndexableField idField = doc.getField( "u" );
946                 if ( idField != null )
947                 {
948                     String id = idField.stringValue();
949                     String[] splits = StringUtils.split( id, "|" );
950                     Dependency dep = new Dependency();
951                     dep.setArtifactId( splits[1] );
952                     dep.setGroupId( splits[0] );
953                     dep.setVersion( splits[2] );
954                     dep.setType( "jar" );
955                     if ( splits.length > 3 && !"NA".equals( splits[3] ) )
956                     {
957                         dep.setClassifier( splits[3] );
958                     }
959                     getLog().info( "found match " + splits[0] + ":" + splits[1] + ":" + splits[2] + " for " + f.getName() );
960                     return dep;
961                 }
962                 else
963                 {
964                     getLog().error( "no idField for " + q );
965                 }
966             }
967             else
968             {
969                 getLog().info( "no repository match for " + f.getName() );
970             }
971         }
972         catch ( Exception x )
973         {
974             getLog().error( x );
975         }
976         return null;
977     }

1250     /**
1251      * Encodes a 128 bit or 160-bit byte array into a String.
1252      *
1253      * @param binaryData Array containing the digest
1254      * @return Encoded hex string, or null if encoding failed
1255      */
1256     static String encode( byte[] binaryData )
1257     {
1258         int bitLength = binaryData.length * 8;
1259         if ( bitLength != 128 && bitLength != 160 )
1260         {
1261             throw new IllegalArgumentException(
1262                 "Unrecognised length for binary data: " + bitLength + " bits" );
1263         }
1264         return String.format( "%0" + bitLength / 4 + "x", new BigInteger( 1, binaryData ) );
1265     }
