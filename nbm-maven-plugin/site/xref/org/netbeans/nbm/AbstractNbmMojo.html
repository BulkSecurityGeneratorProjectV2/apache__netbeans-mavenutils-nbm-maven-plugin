55      static boolean matchesLibrary( Artifact artifact, List<String> libraries, ExamineManifest depExaminator,
56          Log log, boolean useOsgiDependencies )
57      {
58          String artId = artifact.getArtifactId();
59          String grId = artifact.getGroupId();
60          String id = grId + ":" + artId;
61          boolean explicit = libraries.remove( id );
62          if ( explicit )
63          {
64              log.debug(
65                  id + " included as module library, explicitly declared in module descriptor." );
66              return explicit;
67          }
68          if ( Artifact.SCOPE_PROVIDED.equals( artifact.getScope() ) || Artifact.SCOPE_SYSTEM.equals(
69              artifact.getScope() ) )
70          {
71              log.debug(
72                  id + " omitted as module library, has scope 'provided/system'" );
73              return false;
74          }
75          if ( "nbm".equals( artifact.getType() ) )
76          {
77              return false;
78          }
79          if ( depExaminator.isNetBeansModule() || ( useOsgiDependencies && depExaminator.isOsgiBundle() ) )
80          {
81              //TODO I can see how someone might want to include an osgi bundle as library, not dependency.
82              // I guess it won't matter much in 6.9+, in older versions it could be a problem.
83              return false;
84          }
85          log.debug(
86              id + " included as module library, squeezed through all the filters." );
87          return true;
88      }

90      static Dependency resolveNetBeansDependency( Artifact artifact, List<Dependency> deps,
91          ExamineManifest manifest, Log log )
92      {
93          String artId = artifact.getArtifactId();
94          String grId = artifact.getGroupId();
95          String id = grId + ":" + artId;
96          for ( Dependency dep : deps )
97          {
98              if ( id.equals( dep.getId() ) )
99              {
100                 if ( manifest.isNetBeansModule() )
101                 {
102                     return dep;
103                 }
104                 else
105                 {
106                     if ( dep.getExplicitValue() != null )
107                     {
108                         return dep;
109                     }
110                     log.warn(
111                         id + " declared as module dependency in descriptor, but not a NetBeans module" );
112                     return null;
113                 }
114             }
115         }
116         if ( "nbm".equals( artifact.getType() ) )
117         {
118             Dependency dep = new Dependency();
119             dep.setId( id );
120             dep.setType( "spec" );
121             log.debug( "Adding nbm module dependency - " + id );
122             return dep;
123         }
124         if ( manifest.isNetBeansModule() )
125         {
126             Dependency dep = new Dependency();
127             dep.setId( id );
128             dep.setType( "spec" );
129             log.debug( "Adding direct NetBeans module dependency - " + id );
130             return dep;
131         }
132         return null;
133     }

135     protected final NetBeansModule readModuleDescriptor( File descriptor )
136         throws MojoExecutionException
137     {
138         if ( descriptor == null )
139         {
140             throw new MojoExecutionException(
141                 "The module descriptor has to be configured." );
142         }
143         if ( !descriptor.exists() )
144         {
145             throw new MojoExecutionException(
146                 "The module descriptor is missing: '" + descriptor + "'." );
147         }
148         Reader r = null;
149         try
150         {
151             r = new FileReader( descriptor );
152             NetBeansModuleXpp3Reader reader = new NetBeansModuleXpp3Reader();
153             NetBeansModule module = reader.read( r );
154             return module;
155         }
156         catch ( IOException exc )
157         {
158             throw new MojoExecutionException(
159                 "Error while reading module descriptor '" + descriptor + "'.",
160                 exc );
161         }
162         catch ( XmlPullParserException xml )
163         {
164             throw new MojoExecutionException(
165                 "Error while reading module descriptor '" + descriptor + "'.",
166                 xml );
167         }
168         finally
169         {
170             IOUtil.close( r );
171         }
172     }

187                                                List<Artifact> runtimeArtifacts,

193         if ( module != null )

195             List<String> librList = new ArrayList<String>();

199             }

208     static List<ModuleWrapper> getModuleDependencyArtifacts( DependencyNode treeRoot, NetBeansModule module,
209                                                              Dependency[] customDependencies, MavenProject project,
210                                                              Map<Artifact, ExamineManifest> examinerCache,
211                                                              List<Artifact> libraryArtifacts, Log log,
212                                                              boolean useOsgiDependencies )
213         throws MojoExecutionException
214     {
215         List<Dependency> deps = new ArrayList<Dependency>();
216         if (customDependencies != null) {
217             deps.addAll( Arrays.asList( customDependencies ));
218         }
219         if (module != null && !module.getDependencies().isEmpty()) {
220             log.warn( "dependencies in module descriptor are deprecated, use the plugin's parameter moduleDependencies");
221             //we need to make sure a dependency is not twice there, module deps override the config (as is the case with other
222             //configurations)
223             for (Dependency d : module.getDependencies()) {
224                 Dependency found = null;
225                 for (Dependency d2 : deps) {
226                     if (d2.getId().equals(d.getId())) {
227                         found = d2;
228                         break;
229                     }
230                 }
231                 if (found != null) {
232                     deps.remove( found );
233                 }
234                 deps.add(d);
235             }
236         }
237         List<ModuleWrapper> include = new ArrayList<ModuleWrapper>();
238         
239             @SuppressWarnings( "unchecked" )
240             List<Artifact> artifacts = project.getCompileArtifacts();
241             for ( Artifact artifact : artifacts )
242             {
243                 if ( libraryArtifacts.contains( artifact ) )
244                 {
245                     continue;
246                 }
247                 ExamineManifest depExaminator = examinerCache.get( artifact );
248                 if ( depExaminator == null )
249                 {
250                     depExaminator = new ExamineManifest( log );
251                     depExaminator.setArtifactFile( artifact.getFile() );
252                     depExaminator.checkFile();
253                     examinerCache.put( artifact, depExaminator );
254                 }
255                 Dependency dep = resolveNetBeansDependency( artifact, deps, depExaminator, log );
256                 if ( dep != null )
257                 {
258                     ModuleWrapper wr = new ModuleWrapper();
259                     wr.dependency = dep;
260                     wr.artifact = artifact;
261                     wr.transitive = false;
262                     //only direct deps matter to us..
263                     if ( depExaminator.isNetBeansModule() && artifact.getDependencyTrail().size() > 2 )
264                     {
265                         log.debug(
266                             artifact.getId() + " omitted as NetBeans module dependency, not a direct one. Declare it in the pom for inclusion." );
267                         wr.transitive = true;
268 
269                     }
270                     include.add( wr );
271                 }
272                 else
273                 {
274                     if ( useOsgiDependencies && depExaminator.isOsgiBundle() )
275                     {
276                         ModuleWrapper wr = new ModuleWrapper();
277                         wr.osgi = true;
278                         String id = artifact.getGroupId() + ":" + artifact.getArtifactId();
279                         for ( Dependency depe : deps )
280                         {
281                             if ( id.equals( depe.getId() ) )
282                             {
283                                 wr.dependency = depe;
284                             }
285                         }
286                         boolean print = false;
287                         if ( wr.dependency == null )
288                         {
289                             Dependency depe = new Dependency();
290                             depe.setId( id );
291                             depe.setType( "spec" );
292                             wr.dependency = depe;
293                             print = true;
294                         }
295 
296                         wr.artifact = artifact;
297                         wr.transitive = false;
298                         //only direct deps matter to us..
299                         if ( artifact.getDependencyTrail().size() > 2 )
300                         {
301                             log.debug(
302                                 artifact.getId() + " omitted as NetBeans module OSGi dependency, not a direct one. Declare it in the pom for inclusion." );
303                             wr.transitive = true;
304 
305                         }
306                         else
307                         {
308                             if ( print )
309                             {
310                                 log.info( "Adding OSGi bundle dependency - " + id );
311                             }
312                         }
313 
314                         include.add( wr );
315                     }
316                 }
317             }
318         return include;
319     }

335     protected DependencyNode createDependencyTree( MavenProject project, DependencyGraphBuilder dependencyGraphBuilder,
336                                                    String scope )
337         throws MojoExecutionException
338     {
339         ArtifactFilter artifactFilter = createResolvingArtifactFilter( scope );
340         try
341         {
342             return dependencyGraphBuilder.buildDependencyGraph( project, artifactFilter );
343         }
344         catch ( DependencyGraphBuilderException exception )
345         {
346             throw new MojoExecutionException( "Cannot build project dependency tree", exception );
347         }

353      * Gets the artifact filter to use when resolving the dependency tree.
354      *
355      * @return the artifact filter
356      */
357     private ArtifactFilter createResolvingArtifactFilter( String scope )
358     {
359         ArtifactFilter filter;
360 
361         // filter scope
362         if ( scope != null )
363         {
364             getLog().debug( "+ Resolving dependency tree for scope '" + scope + "'" );
365 
366             filter = new ScopeArtifactFilter( scope );
367         }
368         else
369         {
370             filter = null;
371         }
372 
373         return filter;
374     }

376     protected final ArtifactResult turnJarToNbmFile( Artifact art, ArtifactFactory artifactFactory,
377                                                      ArtifactResolver artifactResolver, MavenProject project,
378                                                      ArtifactRepository localRepository )
379         throws MojoExecutionException
380     {
381         if ( "jar".equals( art.getType() ) || "nbm".equals( art.getType() ) )
382         {
383             //TODO, it would be nice to have a check to see if the
384             // "to-be-created" module nbm artifact is actually already in the
385             // list of dependencies (as "nbm-file") or not..
386             // that would be a timesaver
387             ExamineManifest mnf = new ExamineManifest( getLog() );
388             File jar = art.getFile();
389             if ( !jar.isFile() )
390             {
391                 //MNBMODULE-210 with recent CoS changes in netbeans (7.4) jar will be file as we link open projects in the build
392                 // via WorkspaceReader. That's fine here, as all we need is to know if project is osgi or nbm module.
393                 // the nbm file has to be in local repository though.
394                 String path = localRepository.pathOf( art );
395                 File jar2 = new File(localRepository.getBasedir(), path.replace( "/", File.separator));
396                 File manifest = new File(jar, "META-INF/MANIFEST.MF" );
397                 
398                 if (! jar2.isFile() || !manifest.isFile() ) {
399                     getLog().warn( "MNBMODULE-131: need to at least run install phase on " + jar2 );
400                     return new ArtifactResult( null, null );
401                 }
402                 mnf.setManifestFile( manifest );
403             } else {
404                 mnf.setJarFile( jar );
405             }
406             mnf.checkFile();
407             if ( mnf.isNetBeansModule() )
408             {
409                 Artifact nbmArt = artifactFactory.createDependencyArtifact(
410                     art.getGroupId(),
411                     art.getArtifactId(),
412                     art.getVersionRange(),
413                     "nbm-file",
414                     art.getClassifier(),
415                     art.getScope() );
416                 try
417                 {
418                     artifactResolver.resolve( nbmArt, project.getRemoteArtifactRepositories(), localRepository );
419                 }
420 
421                 catch ( ArtifactResolutionException ex )
422                 {
423                     //shall be check before actually resolving from repos?
424                     checkReactor( art, nbmArt );
425                     if ( !nbmArt.isResolved() )
426                     {
427                         throw new MojoExecutionException( "Failed to retrieve the nbm file from repository", ex );
428                     }
429                 }
430                 catch ( ArtifactNotFoundException ex )
431                 {
432                     //shall be check before actually resolving from repos?
433                     checkReactor( art, nbmArt );
434                     if ( !nbmArt.isResolved() )
435                     {
436                         throw new MojoExecutionException( "Failed to retrieve the nbm file from repository", ex );
437                     }
438                 }
439                 return new ArtifactResult( nbmArt, mnf );
440             }
441             if ( mnf.isOsgiBundle() )
442             {
443                 return new ArtifactResult( null, mnf );
444             }
445         }
446         return new ArtifactResult( null, null );
447     }

481     private void checkReactor( Artifact art, Artifact nbmArt )
482     {
483         if ( art.getFile().getName().endsWith( ".jar" ) )
484         {
485             String name = art.getFile().getName();
486             name = name.substring( 0, name.length() - ".jar".length() ) + ".nbm";
487             File fl = new File( art.getFile().getParentFile(), name );
488             if ( fl.exists() )
489             {
490                 nbmArt.setFile( fl );
491                 nbmArt.setResolved( true );
492             }
493         }
494     }
