187     /**
188      * 
189      * @throws MojoExecutionException if an unexpected problem occurs
190      * @throws MojoFailureException if an expected problem occurs
191      */
192     @Override
193     public void execute()
194         throws MojoExecutionException, MojoFailureException
195     {
196         if ( !"nbm-application".equals( project.getPackaging() ) )
197         {
198             throw new MojoExecutionException(
199                 "This goal only makes sense on project with nbm-application packaging." );
200         }
201         Project antProject = antProject();
202         
203         getLog().warn( "WARNING: Unsigned and self-signed WebStart applications are deprecated from JDK7u21 onwards. To ensure future correct functionality please use trusted certificate.");
204 
205         if ( keystore != null && keystorealias != null && keystorepassword != null )
206         {
207             File ks = new File( keystore );
208             if ( !ks.exists() )
209             {
210                 throw new MojoFailureException( "Cannot find keystore file at " + ks.getAbsolutePath() );
211             }
212             else
213             {
214                 //proceed..
215             }
216         }
217         else if ( keystore != null || keystorepassword != null || keystorealias != null )
218         {
219             throw new MojoFailureException(
220                 "If you want to sign the jnlp application, you need to define all three keystore related parameters." );
221         }
222         else
223         {
224             File generatedKeystore = new File( outputDirectory, "generated.keystore" );
225             if ( ! generatedKeystore.exists() )
226             {
227                 getLog().warn( "Keystore related parameters not set, generating a default keystore." );
228                 GenerateKey genTask = (GenerateKey) antProject.createTask( "genkey" );
229                 genTask.setAlias( "jnlp" );
230                 genTask.setStorepass( "netbeans" );
231                 genTask.setDname( "CN=" + System.getProperty( "user.name" ) );
232                 genTask.setKeystore( generatedKeystore.getAbsolutePath() );
233                 genTask.execute();
234             }
235             keystore = generatedKeystore.getAbsolutePath();
236             keystorepassword = "netbeans";
237             keystorealias = "jnlp";
238         }
239 
240         Taskdef taskdef = (Taskdef) antProject.createTask( "taskdef" );
241         taskdef.setClassname( "org.netbeans.nbbuild.MakeJNLP" );
242         taskdef.setName( "makejnlp" );
243         taskdef.execute();
244 
245         taskdef = (Taskdef) antProject.createTask( "taskdef" );
246         taskdef.setClassname( "org.netbeans.nbbuild.VerifyJNLP" );
247         taskdef.setName( "verifyjnlp" );
248         taskdef.execute();
249 
250 
251         try
252         {
253             File webstartBuildDir = new File(
254                 outputDirectory + File.separator + "webstart" + File.separator + brandingToken );
255             if ( webstartBuildDir.exists() )
256             {
257                 FileUtils.deleteDirectory( webstartBuildDir );
258             }
259             webstartBuildDir.mkdirs();
260             final String localCodebase = codebase != null ? codebase : webstartBuildDir.toURI().toString();
261             getLog().info( "Generating webstartable binaries at " + webstartBuildDir.getAbsolutePath() );
262 
263             File nbmBuildDirFile = new File( outputDirectory, brandingToken );
264 
265 //            FileUtils.copyDirectoryStructureIfModified( nbmBuildDirFile, webstartBuildDir );
266 
267             MakeJNLP jnlpTask = (MakeJNLP) antProject.createTask( "makejnlp" );
268             jnlpTask.setDir( webstartBuildDir );
269             jnlpTask.setCodebase( localCodebase );
270             //TODO, how to figure verify excludes..
271             jnlpTask.setVerify( false );
272             jnlpTask.setPermissions( "<security><all-permissions/></security>" );
273             jnlpTask.setSignJars( true );
274 
275             jnlpTask.setAlias( keystorealias );
276             jnlpTask.setKeystore( keystore );
277             jnlpTask.setStorePass( keystorepassword );
278             if ( keystoretype != null )
279             {
280                 jnlpTask.setStoreType( keystoretype );
281             }
282             jnlpTask.setProcessJarVersions( processJarVersions );
283 
284             FileSet fs = jnlpTask.createModules();
285             fs.setDir( nbmBuildDirFile );
286             OrSelector or = new OrSelector();
287             AndSelector and = new AndSelector();
288             FilenameSelector inc = new FilenameSelector();
289             inc.setName( "*/modules/**/*.jar" );
290             or.addFilename( inc );
291             inc = new FilenameSelector();
292             inc.setName( "*/lib/**/*.jar" );
293             or.addFilename( inc );
294             inc = new FilenameSelector();
295             inc.setName( "*/core/**/*.jar" );
296             or.addFilename( inc );
297 
298             ModuleSelector ms = new ModuleSelector();
299             Parameter included = new Parameter();
300             included.setName( "includeClusters" );
301             included.setValue( "" );
302             Parameter excluded = new Parameter();
303             excluded.setName( "excludeClusters" );
304             excluded.setValue( "" );
305             Parameter exModules = new Parameter();
306             exModules.setName( "excludeModules" );
307             exModules.setValue( "" );
308             ms.setParameters( new Parameter[]
309                 {
310                     included,
311                     excluded,
312                     exModules
313                 } );
314             and.add( or );
315             and.add( ms );
316             fs.addAnd( and );
317             jnlpTask.execute();
318 
319             //TODO is it really netbeans/
320             String extSnippet = generateExtensions( fs, antProject, "" ); // "netbeans/"
321 
322             if ( masterJnlpFileName == null )
323             {
324                masterJnlpFileName = brandingToken;
325             }
326 
327             Properties props = new Properties();
328             props.setProperty( "jnlp.codebase", localCodebase );
329             props.setProperty( "app.name", brandingToken );
330             props.setProperty( "app.title", project.getName() );
331             if ( project.getOrganization() != null )
332             {
333                 props.setProperty( "app.vendor", project.getOrganization().getName() );
334             }
335             else
336             {
337                 props.setProperty( "app.vendor", "Nobody" );
338             }
339             String description = project.getDescription() != null ? project.getDescription() : "No Project Description";
340             props.setProperty( "app.description", description );
341             props.setProperty( "branding.token", brandingToken );
342             props.setProperty( "master.jnlp.file.name", masterJnlpFileName );
343             props.setProperty( "netbeans.jnlp.fixPolicy", "false" );
344 
345             StringBuilder stBuilder = new StringBuilder();
346             if ( additionalArguments != null )
347             {
348                 StringTokenizer st = new StringTokenizer( additionalArguments );
349                 while ( st.hasMoreTokens() )
350                 {
351                     String arg = st.nextToken();
352                     if ( arg.startsWith( "-J" ) )
353                     {
354                         if ( stBuilder.length() > 0 )
355                         {
356                             stBuilder.append( ' ' );
357                         }
358                         stBuilder.append( arg.substring( 2 ) );
359                     }
360                 }
361             }
362             props.setProperty( "netbeans.run.params", stBuilder.toString() );
363 
364             File masterJnlp = new File(
365                 webstartBuildDir.getAbsolutePath() + File.separator + masterJnlpFileName + ".jnlp" );
366             filterCopy( masterJnlpFile, "master.jnlp", masterJnlp, props );
367 
368 
369             File startup = copyLauncher( outputDirectory, nbmBuildDirFile );
370             File jnlpDestination = new File(
371                 webstartBuildDir.getAbsolutePath() + File.separator + "startup.jar" );
372 
373             SignJar signTask = (SignJar) antProject.createTask( "signjar" );
374             signTask.setKeystore( keystore );
375             signTask.setStorepass( keystorepassword );
376             signTask.setAlias( keystorealias );
377             if ( keystoretype != null )
378             {
379                 signTask.setStoretype( keystoretype );
380             }
381             signTask.setSignedjar( jnlpDestination );
382             signTask.setJar( startup );
383             signTask.execute();
384 
385             //branding
386             DirectoryScanner ds = new DirectoryScanner();
387             ds.setBasedir( nbmBuildDirFile );
388             ds.setIncludes( new String[]
389                 {
390                     "**/locale/*.jar"
391                 } );
392             ds.scan();
393             String[] includes = ds.getIncludedFiles();
394             StringBuilder brandRefs = new StringBuilder();
395             if ( includes != null && includes.length > 0 )
396             {
397                 File brandingDir = new File( webstartBuildDir, "branding" );
398                 brandingDir.mkdirs();
399                 for ( String incBran : includes )
400                 {
401                     File source = new File( nbmBuildDirFile, incBran );
402                     File dest = new File( brandingDir, source.getName() );
403                     FileUtils.copyFile( source, dest );
404                     brandRefs.append( "    <jar href=\'branding/" ).append( dest.getName() ).append( "\'/>\n" );
405                 }
406 
407                 signTask = (SignJar) antProject.createTask( "signjar" );
408                 signTask.setKeystore( keystore );
409                 signTask.setStorepass( keystorepassword );
410                 signTask.setAlias( keystorealias );
411                 if ( keystoretype != null )
412                 {
413                     signTask.setStoretype( keystoretype );
414                 }
415                 
416                 FileSet set = new FileSet();
417                 set.setDir( brandingDir );
418                 set.setIncludes( "*.jar" );
419                 signTask.addFileset( set );
420                 signTask.execute();
421             }
422 
423             File modulesJnlp = new File(
424                 webstartBuildDir.getAbsolutePath() + File.separator + "modules.jnlp" );
425             props.setProperty( "jnlp.branding.jars", brandRefs.toString() );
426             props.setProperty( "jnlp.resources", extSnippet );
427             filterCopy( null, /* filename is historical */"branding.jnlp", modulesJnlp, props );
428 
429             getLog().info( "Verifying generated webstartable content." );
430             VerifyJNLP verifyTask = (VerifyJNLP) antProject.createTask( "verifyjnlp" );
431             FileSet verify = new FileSet();
432             verify.setFile( masterJnlp );
433             verifyTask.addConfiguredFileset( verify );
434             verifyTask.execute();
435 
436 
437             // create zip archive
438             if ( destinationFile.exists() )
439             {
440                 destinationFile.delete();
441             }
442             ZipArchiver archiver = new ZipArchiver();
443             if ( codebase != null )
444             {
445                 getLog().warn( "Defining <codebase>/${nbm.webstart.codebase} is generally unnecessary" );
446                 archiver.addDirectory( webstartBuildDir );
447             }
448             else
449             {
450                 archiver.addDirectory( webstartBuildDir, null, new String[] { "**/*.jnlp" } );
451                 for ( final File jnlp : webstartBuildDir.listFiles() )
452                 {
453                     if ( !jnlp.getName().endsWith( ".jnlp" ) )
454                     {
455                         continue;
456                     }
457                     archiver.addResource( new PlexusIoResource() {
458                         public @Override InputStream getContents() throws IOException
459                         {
460                             return new ByteArrayInputStream( FileUtils.fileRead( jnlp, "UTF-8" ).replace( localCodebase, "$$codebase" ).getBytes( "UTF-8" ) );
461                         }
462                         public @Override long getLastModified()
463                         {
464                             return jnlp.lastModified();
465                         }
466                         public @Override boolean isExisting()
467                         {
468                             return true;
469                         }
470                         public @Override long getSize()
471                         {
472                             return UNKNOWN_RESOURCE_SIZE;
473                         }
474                         public @Override URL getURL() throws IOException
475                         {
476                             return null;
477                         }
478                         public @Override String getName()
479                         {
480                             return jnlp.getAbsolutePath();
481                         }
482                         public @Override boolean isFile()
483                         {
484                             return true;
485                         }
486                         public @Override boolean isDirectory()
487                         {
488                             return false;
489                         }
490 
491                         @Override
492                         public boolean isSymbolicLink()
493                         {
494                             return false;
495                         }
496                     }, jnlp.getName(), archiver.getDefaultFileMode() );
497                 }
498             }
499             File jdkhome = new File( System.getProperty( "java.home" ) );
500             File servlet = new File( jdkhome, "sample/jnlp/servlet/jnlp-servlet.jar" );
501             if ( ! servlet.isFile() )
502             {
503                 servlet = new File( jdkhome.getParentFile(), "sample/jnlp/servlet/jnlp-servlet.jar" );
504             }
505             if ( servlet.isFile() )
506             {
507                 archiver.addFile( servlet, "WEB-INF/lib/jnlp-servlet.jar" );
508                 archiver.addResource( new PlexusIoResource() {
509                     public @Override InputStream getContents() throws IOException
510                     {
511                         return new ByteArrayInputStream( ( "" +
512                             "<web-app>\n" +
513                             "    <servlet>\n" +
514                             "        <servlet-name>JnlpDownloadServlet</servlet-name>\n" +
515                             "        <servlet-class>jnlp.sample.servlet.JnlpDownloadServlet</servlet-class>\n" +
516                             "    </servlet>\n" +
517                             "    <servlet-mapping>\n" +
518                             "        <servlet-name>JnlpDownloadServlet</servlet-name>\n" +
519                             "        <url-pattern>*.jnlp</url-pattern>\n" +
520                             "    </servlet-mapping>\n" +
521                             "</web-app>\n" ).getBytes() );
522                     }
523                     public @Override long getLastModified()
524                     {
525                         return UNKNOWN_MODIFICATION_DATE;
526                     }
527                     public @Override boolean isExisting()
528                     {
529                         return true;
530                     }
531                     public @Override long getSize()
532                     {
533                         return UNKNOWN_RESOURCE_SIZE;
534                     }
535                     public @Override URL getURL() throws IOException
536                     {
537                         return null;
538                     }
539                     public @Override String getName()
540                     {
541                         return "web.xml";
542                     }
543                     public @Override boolean isFile()
544                     {
545                         return true;
546                     }
547                     public @Override boolean isDirectory()
548                     {
549                         return false;
550                     }
551 
552                     @Override
553                     public boolean isSymbolicLink()
554                     {
555                         return false;
556                     }
557                 }, "WEB-INF/web.xml", archiver.getDefaultFileMode() );
558             }
559             archiver.setDestFile( destinationFile );
560             archiver.createArchive();
561 
562             // attach standalone so that it gets installed/deployed
563             projectHelper.attachArtifact( project, "war", webstartClassifier, destinationFile );
564 
565         }
566         catch ( Exception ex )
567         {
568             throw new MojoExecutionException( "Error creating webstartable binary.", ex );
569         }
570     }

572     /**
573      * @param standaloneBuildDir
574      * @return The name of the jnlp-launcher jarfile in the build directory
575      */
576     private File copyLauncher( File standaloneBuildDir, File builtInstallation )
577         throws IOException
578     {
579         File jnlpStarter =
580             new File( builtInstallation.getAbsolutePath() + File.separator + "harness" + File.separator + "jnlp"
581                 + File.separator + "jnlp-launcher.jar" );
582         // buffer so it isn't reading a byte at a time!
583         InputStream source = null;
584         FileOutputStream outstream = null;
585         try
586         {
587             if ( !jnlpStarter.exists() )
588             {
589                 source = getClass().getClassLoader().getResourceAsStream(
590                     "harness/jnlp/jnlp-launcher.jar" );
591             }
592             else
593             {
594                 source = new FileInputStream( jnlpStarter );
595             }
596             File jnlpDestination = new File(
597                 standaloneBuildDir.getAbsolutePath() + File.separator + "jnlp-launcher.jar" );
598 
599             outstream = new FileOutputStream( jnlpDestination );
600             IOUtil.copy( source, outstream );
601             return jnlpDestination;
602         }
603         finally
604         {
605             IOUtil.close( source );
606             IOUtil.close( outstream );
607         }
608     }

610     private void filterCopy( File sourceFile, String resourcePath, File destinationFile, Properties filterProperties )
611         throws IOException
612     {
613         // buffer so it isn't reading a byte at a time!
614         Reader source = null;
615         Writer destination = null;
616         try
617         {
618             InputStream instream;
619             if ( sourceFile != null )
620             {
621                 instream = new FileInputStream( sourceFile );
622             }
623             else
624             {
625                 instream = getClass().getClassLoader().getResourceAsStream( resourcePath );
626             }
627             FileOutputStream outstream = new FileOutputStream( destinationFile );
628 
629             source = new BufferedReader( new InputStreamReader( instream, "UTF-8" ) );
630             destination = new OutputStreamWriter( outstream, "UTF-8" );
631 
632             // support ${token}
633             Reader reader = new InterpolationFilterReader( source, filterProperties, "${", "}" );
634 
635             IOUtil.copy( reader, destination );
636         }
637         finally
638         {
639             IOUtil.close( source );
640             IOUtil.close( destination );
641         }
642     }

644     /**
645      * copied from MakeMasterJNLP ant task.
646      * @param files
647      * @param antProject
648      * @param masterPrefix
649      * @return
650      * @throws java.io.IOException
651      */
652     private String generateExtensions( FileSet files, Project antProject, String masterPrefix )
653         throws IOException
654     {
655         StringBuilder buff = new StringBuilder();
656         for ( String nm : files.getDirectoryScanner( antProject ).getIncludedFiles() )
657         {
658             File jar = new File( files.getDir( antProject ), nm );
659 
660             if ( !jar.canRead() )
661             {
662                 throw new IOException( "Cannot read file: " + jar );
663             }
664 
665             JarFile theJar = new JarFile( jar );
666             Attributes attr = theJar.getManifest().getMainAttributes();
667             String codenamebase = attr.getValue( "OpenIDE-Module" );
668             if ( codenamebase == null )
669             {
670                 codenamebase = attr.getValue("Bundle-SymbolicName");
671             }
672             if ( codenamebase == null )
673             {
674                 throw new IOException( "Not a NetBeans Module: " + jar );
675             }
676             
677             // see http://hg.netbeans.org/main-silver/rev/87823abb86d9
678             if (codenamebase.equals("org.objectweb.asm.all")
679                     && jar.getParentFile().getName().equals("core")
680                     && jar.getParentFile().getParentFile().getName().startsWith("platform")) {
681                 continue;
682             }
683             {
684                 int slash = codenamebase.indexOf( '/' );
685                 if ( slash >= 0 )
686                 {
687                     codenamebase = codenamebase.substring( 0, slash );
688                 }
689             }
690             String dashcnb = codenamebase.replace( '.', '-' );
691 
692             buff.append( "    <extension name='" ).append( codenamebase ).append( "' href='" ).append( masterPrefix ).append( dashcnb ).append( ".jnlp' />\n" );
693             theJar.close();
694         }
695         return buff.toString();
696 
697     }
